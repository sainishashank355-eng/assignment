package java1;
import java.util.*;
public class cal {

	public static void main(String[] args) {
		Scanner os= new Scanner(System.in);
	    System.out.println("Enter 2 Number for perfomation of logic operator and also give the operator");
		int a=os.nextInt();
		int b=os.nextInt();
		char c=os.next().charAt(0);
	    switch(c){
	    		case '+':
		    		System.out.print("Addtion =");
		    		System.out.print(a+b);
		    		break;
	    		case '-':
	    			System.out.print("Substraction =");
		    		System.out.print(a-b);
		    		break;
	    		case '*':
	    			System.out.print("Multiplication =");
		    		System.out.print(a*b);
		    		break;
	    		case '/':
	    			System.out.print("Dividtion {quotient} =");
		    		System.out.print(a/b);
		    		break;
	    		case '%':
	    			System.out.print("Division{Remainder} =");
		    		System.out.print(a%b);
		    		break;
	    		default :
		    		System.out.println("Not Valid");
		    		break;
	    }

}
}
